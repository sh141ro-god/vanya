CREATE TABLE Role (
    Role_ID     INT             NOT NULL IDENTITY(1,1),
    Name        VARCHAR(25)     NOT NULL,
    Description VARCHAR(100)        NULL,

    CONSTRAINT PK_Role      PRIMARY KEY (Role_ID),
    CONSTRAINT UQ_Role_Name UNIQUE (Name)
);
GO

CREATE TABLE Employees (
    Employees_ID    INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Password        VARCHAR(256)    NOT NULL,
    Role            INT                 NULL,
    Login           VARCHAR(25)     NOT NULL,

    CONSTRAINT PK_Employees       PRIMARY KEY (Employees_ID),
    CONSTRAINT UQ_Employees_Login UNIQUE (Login),

    CONSTRAINT FK_Employees_Role
        FOREIGN KEY (Role) REFERENCES Role (Role_ID)
        ON DELETE SET NULL
        ON UPDATE NO ACTION
);
GO

CREATE TABLE Client (
    Client_ID   INT             NOT NULL IDENTITY(1,1),
    Name        VARCHAR(25)     NOT NULL,
    Number      VARCHAR(12)         NULL,
    Address     VARCHAR(50)         NULL,
    Email       VARCHAR(50)         NULL,

    CONSTRAINT PK_Client       PRIMARY KEY (Client_ID),
    CONSTRAINT UQ_Client_Email UNIQUE (Email)
);
GO

CREATE TABLE OrderStatus (
    OrderStatus_ID  INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Description     VARCHAR(100)        NULL,

    CONSTRAINT PK_OrderStatus      PRIMARY KEY (OrderStatus_ID),
    CONSTRAINT UQ_OrderStatus_Name UNIQUE (Name)
);
GO

CREATE TABLE Restaurant (
    Restaurant_ID   INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Numder          VARCHAR(12)         NULL,
    Address         VARCHAR(50)         NULL,

    CONSTRAINT PK_Restaurant PRIMARY KEY (Restaurant_ID)
);
GO

CREATE TABLE DishCategory (
    DishCategory_ID INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Description     VARCHAR(100)        NULL,

    CONSTRAINT PK_DishCategory      PRIMARY KEY (DishCategory_ID),
    CONSTRAINT UQ_DishCategory_Name UNIQUE (Name)
);
GO

CREATE TABLE Dish (
    Dish_ID         INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Restaurant      INT             NOT NULL,
    Description     VARCHAR(100)        NULL,
    DishCategory    INT                 NULL,
    Cost            INT             NOT NULL,

    CONSTRAINT PK_Dish      PRIMARY KEY (Dish_ID),
    CONSTRAINT CK_Dish_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Dish_Restaurant
        FOREIGN KEY (Restaurant) REFERENCES Restaurant (Restaurant_ID)
        ON DELETE NO ACTION
        ON UPDATE NO ACTION,

    CONSTRAINT FK_Dish_DishCategory
        FOREIGN KEY (DishCategory) REFERENCES DishCategory (DishCategory_ID)
        ON DELETE SET NULL
        ON UPDATE NO ACTION
);
GO

CREATE TABLE [Order] (
    Order_ID    INT             NOT NULL IDENTITY(1,1),
    Employee    INT                 NULL,
    Client      INT                 NULL,
    Status      INT                 NULL,
    Address     VARCHAR(50)     NOT NULL,
    Cost        INT                 NULL,

    CONSTRAINT PK_Order      PRIMARY KEY (Order_ID),
    CONSTRAINT CK_Order_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Order_Employee
        FOREIGN KEY (Employee) REFERENCES Employees (Employees_ID)
        ON DELETE SET NULL
        ON UPDATE NO ACTION,

    CONSTRAINT FK_Order_Client
        FOREIGN KEY (Client) REFERENCES Client (Client_ID)
        ON DELETE SET NULL
        ON UPDATE NO ACTION,

    CONSTRAINT FK_Order_Status
        FOREIGN KEY (Status) REFERENCES OrderStatus (OrderStatus_ID)
        ON DELETE SET NULL
        ON UPDATE NO ACTION
);
GO

CREATE TABLE OrderPart (
    Dish_ID     INT     NOT NULL,
    Order_ID    INT     NOT NULL,
    Count       INT     NOT NULL DEFAULT 1,
    Cost        INT     NOT NULL,

    CONSTRAINT PK_OrderPart      PRIMARY KEY (Dish_ID, Order_ID),
    CONSTRAINT CK_OrderPart_Cnt  CHECK (Count > 0),
    CONSTRAINT CK_OrderPart_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_OrderPart_Dish
        FOREIGN KEY (Dish_ID) REFERENCES Dish (Dish_ID)
        ON DELETE NO ACTION
        ON UPDATE NO ACTION,

    CONSTRAINT FK_OrderPart_Order
        FOREIGN KEY (Order_ID) REFERENCES [Order] (Order_ID)
        ON DELETE CASCADE
        ON UPDATE NO ACTION
);
GO
