
CREATE TABLE Role (
    Role_ID     INT             NOT NULL IDENTITY(1,1),
    Name        VARCHAR(25)     NOT NULL,
    Description VARCHAR(100)    NOT NULL DEFAULT '',

    CONSTRAINT PK_Role      PRIMARY KEY (Role_ID),
    CONSTRAINT UQ_Role_Name UNIQUE (Name)
);



CREATE TABLE Employees (
    Employees_ID    INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Password        VARCHAR(255)    NOT NULL,
    Login           VARCHAR(25)     NOT NULL,
    Role_ID         INT                 NULL,

    CONSTRAINT PK_Employees       PRIMARY KEY (Employees_ID),
    CONSTRAINT UQ_Employees_Login UNIQUE (Login),

    CONSTRAINT FK_Employees_Role
        FOREIGN KEY (Role_ID) REFERENCES Role (Role_ID)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);


CREATE TABLE Order_Status (
    Order_Status_ID INT             NOT NULL IDENTITY(1,1),
    Name            VARCHAR(25)     NOT NULL,
    Description     VARCHAR(100)    NOT NULL DEFAULT '',

    CONSTRAINT PK_Order_Status      PRIMARY KEY (Order_Status_ID),
    CONSTRAINT UQ_Order_Status_Name UNIQUE (Name)
);



CREATE TABLE Company (
    Company_ID   INT             NOT NULL IDENTITY(1,1),
    Name        VARCHAR(25)     NOT NULL,
    Number      VARCHAR(12)         NULL,
    Address     VARCHAR(50)         NULL,
    Email       VARCHAR(25)         NULL,

    CONSTRAINT PK_Company       PRIMARY KEY (Company_ID),
    CONSTRAINT UQ_Company_Email UNIQUE (Email)
);


CREATE TABLE Contract (
    Contract_ID     INT             NOT NULL IDENTITY(1,1),
    Client          INT             NOT NULL,
    DeliveryTime    DATETIME        NOT NULL,
    Cost            INT             NOT NULL DEFAULT 0,
    StartDate       DATETIME        NOT NULL DEFAULT GETDATE(),
    EndDate         DATETIME            NULL,

    CONSTRAINT PK_Contract   PRIMARY KEY (Contract_ID),
    CONSTRAINT CK_Contract_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Contract_Client
        FOREIGN KEY (Client) REFERENCES Company (Company_ID)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);



CREATE TABLE [Order] (
    Order_ID    INT     NOT NULL IDENTITY(1,1),
    Contract    INT     NOT NULL,
    Employee    INT         NULL,
    Status      INT     NOT NULL,

    CONSTRAINT PK_Order PRIMARY KEY (Order_ID),

    CONSTRAINT FK_Order_Contract
        FOREIGN KEY (Contract) REFERENCES Contract (Contract_ID)
        ON DELETE NO ACTION
        ON UPDATE CASCADE,

    CONSTRAINT FK_Order_Employee
        FOREIGN KEY (Employee) REFERENCES Employees (Employees_ID)
        ON DELETE SET NULL
        ON UPDATE CASCADE,

    CONSTRAINT FK_Order_Status
        FOREIGN KEY (Status) REFERENCES Order_Status (Order_Status_ID)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);



CREATE TABLE Категория_блюд (
    ID_категории    INT             NOT NULL IDENTITY(1,1),
    Название        VARCHAR(25)     NOT NULL,
    Описание        VARCHAR(100)        NULL,

    CONSTRAINT PK_Категория_блюд      PRIMARY KEY (ID_категории),
    CONSTRAINT UQ_Категория_блюд_Назв UNIQUE (Название)
);


CREATE TABLE Dish (
    Dish_ID     INT             NOT NULL IDENTITY(1,1),
    Name        VARCHAR(25)     NOT NULL,
    Cost        INT             NOT NULL DEFAULT 0,
    Description VARCHAR(100)        NULL,
    Category_ID INT                 NULL,

    CONSTRAINT PK_Dish      PRIMARY KEY (Dish_ID),
    CONSTRAINT CK_Dish_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Dish_Category
        FOREIGN KEY (Category_ID) REFERENCES Категория_блюд (ID_категории)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);



CREATE TABLE Order_Part (
    Order_ID    INT     NOT NULL,
    Dish_ID     INT     NOT NULL,
    Count       INT     NOT NULL DEFAULT 1,
    Cost        INT     NOT NULL DEFAULT 0,

    CONSTRAINT PK_Order_Part      PRIMARY KEY (Order_ID, Dish_ID),
    CONSTRAINT CK_Order_Part_Cnt  CHECK (Count > 0),
    CONSTRAINT CK_Order_Part_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Order_Part_Order
        FOREIGN KEY (Order_ID) REFERENCES [Order] (Order_ID)
        ON DELETE CASCADE
        ON UPDATE CASCADE,

    CONSTRAINT FK_Order_Part_Dish
        FOREIGN KEY (Dish_ID) REFERENCES Dish (Dish_ID)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);

SELECT
    o.Order_ID,
    c.Name          AS Клиент,
    e.Name          AS Сотрудник,
    os.Name         AS Статус,
    ct.DeliveryTime AS Время_доставки,
    ct.Cost         AS Стоимость_контракта
FROM [Order] o
JOIN Contract    ct ON ct.Contract_ID     = o.Contract
JOIN Company       c ON c.Company_ID        = ct.Client
JOIN Order_Status os ON os.Order_Status_ID = o.Status
LEFT JOIN Employees e ON e.Employees_ID   = o.Employee;


SELECT
    op.Order_ID     AS Заказ,
    d.Name          AS Блюдо,
    op.Count        AS Количество,
    op.Cost         AS Сумма
FROM Order_Part op
JOIN Dish d ON d.Dish_ID = op.Dish_ID
ORDER BY op.Order_ID;

-- Меню по категориям
SELECT
    k.Название      AS Категория,
    d.Name          AS Блюдо,
    d.Cost          AS Цена
FROM Dish d
LEFT JOIN Категория_блюд k ON k.ID_категории = d.Category_ID
ORDER BY k.Название;
