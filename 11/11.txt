
CREATE TABLE Role (
    Role_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL UNIQUE,
    Description VARCHAR(100) NOT NULL DEFAULT ''
);


CREATE TABLE Employee (
    Employee_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL,
    Password VARCHAR(255) NOT NULL,
    Login VARCHAR(25) NOT NULL UNIQUE,
    Role_ID INT NULL,

    CONSTRAINT FK_Employee_Role
        FOREIGN KEY (Role_ID) REFERENCES Role(Role_ID)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);


CREATE TABLE Company (
    Company_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL,
    Number VARCHAR(12) NULL,
    Address VARCHAR(50) NULL,
    Email VARCHAR(25) NULL UNIQUE
);


CREATE TABLE Company_Employees (
    Company_ID INT NOT NULL,
    Employee_ID INT NOT NULL,

    PRIMARY KEY (Company_ID, Employee_ID),

    CONSTRAINT FK_CE_Company
        FOREIGN KEY (Company_ID) REFERENCES Company(Company_ID)
        ON DELETE CASCADE,

    CONSTRAINT FK_CE_Employee
        FOREIGN KEY (Employee_ID) REFERENCES Employee(Employee_ID)
        ON DELETE CASCADE
);


CREATE TABLE Contract (
    Contract_ID INT IDENTITY(1,1) PRIMARY KEY,
    Company_ID INT NOT NULL,
    DeliveryTime DATETIME NOT NULL,
    Cost INT NOT NULL DEFAULT 0,
    StartDate DATETIME NOT NULL DEFAULT GETDATE(),
    EndDate DATETIME NULL,

    CONSTRAINT CK_Contract_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Contract_Company
        FOREIGN KEY (Company_ID) REFERENCES Company(Company_ID)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);


CREATE TABLE Order_Status (
    Order_Status_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL UNIQUE,
    Description VARCHAR(100) NOT NULL DEFAULT ''
);


CREATE TABLE [Order] (
    Order_ID INT IDENTITY(1,1) PRIMARY KEY,
    Contract_ID INT NOT NULL,
    Employee_ID INT NULL,
    Status_ID INT NOT NULL,

    CONSTRAINT FK_Order_Contract
        FOREIGN KEY (Contract_ID) REFERENCES Contract(Contract_ID)
        ON DELETE NO ACTION,

    CONSTRAINT FK_Order_Employee
        FOREIGN KEY (Employee_ID) REFERENCES Employee(Employee_ID)
        ON DELETE SET NULL,

    CONSTRAINT FK_Order_Status
        FOREIGN KEY (Status_ID) REFERENCES Order_Status(Order_Status_ID)
        ON DELETE NO ACTION
);


CREATE TABLE Category (
    Category_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL UNIQUE,
    Description VARCHAR(100) NULL
);


CREATE TABLE Dish (
    Dish_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(25) NOT NULL,
    Cost INT NOT NULL DEFAULT 0,
    Description VARCHAR(100) NULL,
    Category_ID INT NULL,

    CONSTRAINT CK_Dish_Cost CHECK (Cost >= 0),

    CONSTRAINT FK_Dish_Category
        FOREIGN KEY (Category_ID) REFERENCES Category(Category_ID)
        ON DELETE SET NULL
);


CREATE TABLE Ingredient (
    Ingredient_ID INT IDENTITY(1,1) PRIMARY KEY,
    Name VARCHAR(50) NOT NULL
);


CREATE TABLE Dish_Ingredient (
    Dish_ID INT NOT NULL,
    Ingredient_ID INT NOT NULL,
    Count INT NOT NULL DEFAULT 1,

    PRIMARY KEY (Dish_ID, Ingredient_ID),

    CONSTRAINT FK_DI_Dish
        FOREIGN KEY (Dish_ID) REFERENCES Dish(Dish_ID)
        ON DELETE CASCADE,

    CONSTRAINT FK_DI_Ingredient
        FOREIGN KEY (Ingredient_ID) REFERENCES Ingredient(Ingredient_ID)
        ON DELETE CASCADE
);


CREATE TABLE Order_Part (
    Order_ID INT NOT NULL,
    Dish_ID INT NOT NULL,
    Count INT NOT NULL DEFAULT 1,
    Cost INT NOT NULL DEFAULT 0,

    PRIMARY KEY (Order_ID, Dish_ID),

    CONSTRAINT CK_OrderPart_Count CHECK (Count > 0),
    CONSTRAINT CK_OrderPart_Cost CHECK (Cost >= 0),

    FOREIGN KEY (Order_ID) REFERENCES [Order](Order_ID) ON DELETE CASCADE,
    FOREIGN KEY (Dish_ID) REFERENCES Dish(Dish_ID)
);
