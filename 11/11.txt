-- ============================================================
--  СОЗДАНИЕ ТАБЛИЦ
-- ============================================================

-- 1. Роль (должность сотрудника доставки)
CREATE TABLE Role (
    Role_ID     INT           PRIMARY KEY IDENTITY(1,1),
    Name        VARCHAR(25)   NOT NULL,
    Description VARCHAR(100)  NULL
);
GO

-- 2. Сотрудник (сотрудник службы доставки)
CREATE TABLE Employees (
    Employees_ID INT          PRIMARY KEY IDENTITY(1,1),
    Name         VARCHAR(50)  NOT NULL,
    Password     VARCHAR(25)  NOT NULL,
    Login        VARCHAR(25)  NOT NULL UNIQUE,
    Role_ID      INT          NOT NULL,
    CONSTRAINT FK_Employees_Role FOREIGN KEY (Role_ID) REFERENCES Role(Role_ID)
);
GO

-- 3. Статус заказа
CREATE TABLE Order_Status (
    Order_Status_ID INT          PRIMARY KEY IDENTITY(1,1),
    Name            VARCHAR(25)  NOT NULL,
    Description     VARCHAR(100) NULL
);
GO

-- 4. Компания-клиент
CREATE TABLE Company (
    Company_ID  INT          PRIMARY KEY IDENTITY(1,1),
    Name        VARCHAR(50)  NOT NULL,
    Number      VARCHAR(12)  NULL,
    Address     VARCHAR(100) NULL,
    Email       VARCHAR(50)  NULL
);
GO

-- 5. Договор
CREATE TABLE Contract (
    Contract_ID  INT      PRIMARY KEY IDENTITY(1,1),
    Company_ID   INT      NOT NULL,
    DeliveryTime TIME     NOT NULL,
    StartDate    DATE     NOT NULL,
    EndDate      DATE     NOT NULL,
    CONSTRAINT FK_Contract_Company FOREIGN KEY (Company_ID) REFERENCES Company(Company_ID)
);
GO

-- 6. Сотрудник компании-клиента
CREATE TABLE Company_Employees (
    CompanyEmployees_ID INT          PRIMARY KEY IDENTITY(1,1),
    Name                VARCHAR(50)  NOT NULL,
    Password            VARCHAR(25)  NOT NULL,
    Login               VARCHAR(25)  NOT NULL UNIQUE,
    Company_ID          INT          NOT NULL,
    CONSTRAINT FK_CompanyEmp_Company FOREIGN KEY (Company_ID) REFERENCES Company(Company_ID)
);
GO

-- 7. Заказ
CREATE TABLE [Order] (
    Order_ID    INT  PRIMARY KEY IDENTITY(1,1),
    Contract_ID INT  NOT NULL,
    Employee_ID INT  NOT NULL,   -- доставщик
    Status_ID   INT  NOT NULL,
    OrderDate   DATE NOT NULL DEFAULT CAST(GETDATE() AS DATE),
    CONSTRAINT FK_Order_Contract FOREIGN KEY (Contract_ID) REFERENCES Contract(Contract_ID),
    CONSTRAINT FK_Order_Employee FOREIGN KEY (Employee_ID) REFERENCES Employees(Employees_ID),
    CONSTRAINT FK_Order_Status   FOREIGN KEY (Status_ID)   REFERENCES Order_Status(Order_Status_ID)
);
GO

-- 8. Связь «сотрудник компании — заказ» (кто из компании сделал заказ)
CREATE TABLE Company_Employees_Order (
    CompanyEmployees_ID INT NOT NULL,
    Order_ID            INT NOT NULL,
    CONSTRAINT PK_CompEmpOrder PRIMARY KEY (CompanyEmployees_ID, Order_ID),
    CONSTRAINT FK_CEO_CompEmp  FOREIGN KEY (CompanyEmployees_ID) REFERENCES Company_Employees(CompanyEmployees_ID),
    CONSTRAINT FK_CEO_Order    FOREIGN KEY (Order_ID)            REFERENCES [Order](Order_ID)
);
GO

-- 9. Категория блюд
CREATE TABLE Category (
    Category_ID INT          PRIMARY KEY IDENTITY(1,1),
    Name        VARCHAR(50)  NOT NULL,
    Description VARCHAR(100) NULL
);
GO

-- 10. Блюдо
CREATE TABLE Dish (
    Dish_ID     INT           PRIMARY KEY IDENTITY(1,1),
    Name        VARCHAR(50)   NOT NULL,
    Cost        DECIMAL(10,2) NOT NULL,
    Description VARCHAR(200)  NULL,
    Category_ID INT           NOT NULL,
    CONSTRAINT FK_Dish_Category FOREIGN KEY (Category_ID) REFERENCES Category(Category_ID)
);
GO

-- 11. Позиция в заказе (что именно заказано)
CREATE TABLE Order_Part (
    CompanyEmployees_ID INT           NOT NULL,
    Dish_ID             INT           NOT NULL,
    Count               INT           NOT NULL DEFAULT 1,
    Cost                DECIMAL(10,2) NOT NULL,
    CONSTRAINT PK_OrderPart         PRIMARY KEY (CompanyEmployees_ID, Dish_ID),
    CONSTRAINT FK_OP_CompEmp        FOREIGN KEY (CompanyEmployees_ID) REFERENCES Company_Employees(CompanyEmployees_ID),
    CONSTRAINT FK_OP_Dish           FOREIGN KEY (Dish_ID)             REFERENCES Dish(Dish_ID),
    CONSTRAINT CHK_OP_Count         CHECK (Count > 0)
);
GO

-- 12. Ингредиент
CREATE TABLE Ingredient (
    Ingredient_ID INT         PRIMARY KEY IDENTITY(1,1),
    Name          VARCHAR(50) NOT NULL
);
GO

-- 13. Состав блюда (блюдо ↔ ингредиент)
CREATE TABLE Dish_Ingredient (
    Dish_ID       INT           NOT NULL,
    Ingredient_ID INT           NOT NULL,
    Count         INT           NOT NULL DEFAULT 1,
    CONSTRAINT PK_DishIngredient    PRIMARY KEY (Dish_ID, Ingredient_ID),
    CONSTRAINT FK_DI_Dish           FOREIGN KEY (Dish_ID)       REFERENCES Dish(Dish_ID),
    CONSTRAINT FK_DI_Ingredient     FOREIGN KEY (Ingredient_ID) REFERENCES Ingredient(Ingredient_ID),
    CONSTRAINT CHK_DI_Count         CHECK (Count > 0)
);
GO

-- 1. Все заказы с договором, компанией, статусом и курьером
SELECT
    o.Order_ID        AS [Заказ №],
    o.OrderDate       AS [Дата],
    c.Name            AS [Компания],
    os.Name           AS [Статус],
    e.Name            AS [Курьер],
    ct.DeliveryTime   AS [Время доставки]
FROM [Order] o
JOIN Contract    ct ON o.Contract_ID = ct.Contract_ID
JOIN Company      c ON ct.Company_ID  = c.Company_ID
JOIN Order_Status os ON o.Status_ID   = os.Order_Status_ID
JOIN Employees    e  ON o.Employee_ID = e.Employees_ID
ORDER BY o.Order_ID;
GO

-- 2. Состав и стоимость каждого заказа
SELECT
    ceo.Order_ID             AS [Заказ №],
    ce.Name                  AS [Сотрудник компании],
    d.Name                   AS [Блюдо],
    op.Count                 AS [Кол-во],
    op.Cost                  AS [Сумма]
FROM Order_Part op
JOIN Company_Employees ce      ON op.CompanyEmployees_ID = ce.CompanyEmployees_ID
JOIN Company_Employees_Order ceo ON ceo.CompanyEmployees_ID = op.CompanyEmployees_ID
JOIN Dish d                    ON op.Dish_ID = d.Dish_ID
ORDER BY ceo.Order_ID, op.Dish_ID;
GO

-- 3. Итоговая сумма по каждому заказу
SELECT
    ceo.Order_ID   AS [Заказ №],
    SUM(op.Cost)   AS [Итого, руб.]
FROM Order_Part op
JOIN Company_Employees_Order ceo ON ceo.CompanyEmployees_ID = op.CompanyEmployees_ID
GROUP BY ceo.Order_ID
ORDER BY ceo.Order_ID;
GO
