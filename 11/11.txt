USE master;
GO

IF EXISTS (SELECT name FROM sys.databases WHERE name = 'delivery_db')
    DROP DATABASE delivery_db;
GO

CREATE DATABASE delivery_db
    COLLATE Cyrillic_General_CI_AS;
GO

USE delivery_db;
GO

CREATE TABLE Должность (
    ID_должности    INT             NOT NULL IDENTITY(1,1),
    Название        VARCHAR(25)     NOT NULL,
    Описание        VARCHAR(100)        NULL,

    CONSTRAINT PK_Должность      PRIMARY KEY (ID_должности),
    CONSTRAINT UQ_Должность_Назв UNIQUE (Название)
);
GO

CREATE TABLE Сотрудник (
    ID_сотрудника   INT             NOT NULL IDENTITY(1,1),
    ФИО             VARCHAR(25)     NOT NULL,
    Пароль          VARCHAR(255)    NOT NULL,
    Логин           VARCHAR(25)     NOT NULL,
    Должность       INT                 NULL,

    CONSTRAINT PK_Сотрудник       PRIMARY KEY (ID_сотрудника),
    CONSTRAINT UQ_Сотрудник_Логин UNIQUE (Логин),

    CONSTRAINT FK_Сотрудник_Должность
        FOREIGN KEY (Должность) REFERENCES Должность (ID_должности)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);
GO

CREATE TABLE Клиент (
    ID_клиента  INT             NOT NULL IDENTITY(1,1),
    Название    VARCHAR(25)     NOT NULL,
    Телефон     VARCHAR(12)         NULL,
    Адрес       VARCHAR(50)         NULL,
    Email       VARCHAR(25)         NULL,

    CONSTRAINT PK_Клиент       PRIMARY KEY (ID_клиента),
    CONSTRAINT UQ_Клиент_Email UNIQUE (Email)
);
GO

CREATE TABLE Статус_заказа (
    ID_статуса  INT             NOT NULL IDENTITY(1,1),
    Название    VARCHAR(25)     NOT NULL,
    Описание    VARCHAR(100)        NULL,

    CONSTRAINT PK_Статус_заказа      PRIMARY KEY (ID_статуса),
    CONSTRAINT UQ_Статус_заказа_Назв UNIQUE (Название)
);
GO

CREATE TABLE Договор (
    ID_договора     INT             NOT NULL IDENTITY(1,1),
    Клиент          INT             NOT NULL,
    Время_доставки  DATETIME        NOT NULL,
    Дата_начала     DATETIME        NOT NULL DEFAULT GETDATE(),
    Дата_окончания  DATETIME            NULL,

    CONSTRAINT PK_Договор PRIMARY KEY (ID_договора),

    CONSTRAINT FK_Договор_Клиент
        FOREIGN KEY (Клиент) REFERENCES Клиент (ID_клиента)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);
GO

CREATE TABLE Заказ (
    ID_заказа   INT     NOT NULL IDENTITY(1,1),
    Договор     INT     NOT NULL,
    Сотрудник   INT         NULL,
    Статус      INT     NOT NULL,
    Стоимость   INT     NOT NULL DEFAULT 0,

    CONSTRAINT PK_Заказ PRIMARY KEY (ID_заказа),
    CONSTRAINT CK_Заказ_Стоим CHECK (Стоимость >= 0),

    CONSTRAINT FK_Заказ_Договор
        FOREIGN KEY (Договор) REFERENCES Договор (ID_договора)
        ON DELETE NO ACTION
        ON UPDATE CASCADE,

    CONSTRAINT FK_Заказ_Сотрудник
        FOREIGN KEY (Сотрудник) REFERENCES Сотрудник (ID_сотрудника)
        ON DELETE SET NULL
        ON UPDATE CASCADE,

    CONSTRAINT FK_Заказ_Статус
        FOREIGN KEY (Статус) REFERENCES Статус_заказа (ID_статуса)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);
GO

CREATE TABLE Категория_блюд (
    ID_категории    INT             NOT NULL IDENTITY(1,1),
    Название        VARCHAR(25)     NOT NULL,
    Описание        VARCHAR(100)        NULL,

    CONSTRAINT PK_Категория_блюд      PRIMARY KEY (ID_категории),
    CONSTRAINT UQ_Категория_блюд_Назв UNIQUE (Название)
);
GO

CREATE TABLE Блюдо (
    ID_блюда    INT             NOT NULL IDENTITY(1,1),
    Название    VARCHAR(25)     NOT NULL,
    Цена        INT             NOT NULL DEFAULT 0,
    Описание    VARCHAR(100)        NULL,
    Категория   INT                 NULL,

    CONSTRAINT PK_Блюдо      PRIMARY KEY (ID_блюда),
    CONSTRAINT CK_Блюдо_Цена CHECK (Цена >= 0),

    CONSTRAINT FK_Блюдо_Категория
        FOREIGN KEY (Категория) REFERENCES Категория_блюд (ID_категории)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);
GO

CREATE TABLE Позиция_в_заказе (
    Заказ   INT     NOT NULL,
    Блюдо   INT     NOT NULL,
    Количество INT  NOT NULL DEFAULT 1,
    Цена    INT     NOT NULL DEFAULT 0,

    CONSTRAINT PK_Позиция_в_заказе      PRIMARY KEY (Заказ, Блюдо),
    CONSTRAINT CK_Позиция_Кол           CHECK (Количество > 0),
    CONSTRAINT CK_Позиция_Цена          CHECK (Цена >= 0),

    CONSTRAINT FK_Позиция_Заказ
        FOREIGN KEY (Заказ) REFERENCES Заказ (ID_заказа)
        ON DELETE CASCADE
        ON UPDATE CASCADE,

    CONSTRAINT FK_Позиция_Блюдо
        FOREIGN KEY (Блюдо) REFERENCES Блюдо (ID_блюда)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);
GO

CREATE TABLE Продукт (
    ID_продукта INT             NOT NULL IDENTITY(1,1),
    Название    VARCHAR(25)     NOT NULL,

    CONSTRAINT PK_Продукт      PRIMARY KEY (ID_продукта),
    CONSTRAINT UQ_Продукт_Назв UNIQUE (Название)
);
GO

CREATE TABLE Состав (
    Продукт     INT     NOT NULL,
    Блюдо       INT     NOT NULL,
    Количество  INT     NOT NULL DEFAULT 1,

    CONSTRAINT PK_Состав     PRIMARY KEY (Продукт, Блюдо),
    CONSTRAINT CK_Состав_Кол CHECK (Количество > 0),

    CONSTRAINT FK_Состав_Продукт
        FOREIGN KEY (Продукт) REFERENCES Продукт (ID_продукта)
        ON DELETE NO ACTION
        ON UPDATE CASCADE,

    CONSTRAINT FK_Состав_Блюдо
        FOREIGN KEY (Блюдо) REFERENCES Блюдо (ID_блюда)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
GO
